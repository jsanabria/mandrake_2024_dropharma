CREATE PROCEDURE sp_mdk_asiento_venta_salida(IN p_salida_id INT UNSIGNED, IN p_centro_costo_id BIGINT)
BEGIN
  DECLARE v_fecha DATE;
  DECLARE v_cliente INT;
  DECLARE v_nro VARCHAR(60);
  DECLARE v_moneda VARCHAR(6);
  DECLARE v_tasa DECIMAL(20,6);
  DECLARE v_total_bs DECIMAL(18,2);
  DECLARE v_total_trx DECIMAL(18,2);
  DECLARE v_username VARCHAR(30);
  DECLARE v_asiento_id BIGINT;

  DECLARE v_cta_cxc BIGINT;
  DECLARE v_cta_ventas BIGINT;

  -- Cuentas desde configuración
  SET v_cta_cxc = fn_cta_id_mdk('CTA_CXC');
  SET v_cta_ventas = fn_cta_id_mdk('CTA_VENTAS');

  IF v_cta_cxc IS NULL OR v_cta_ventas IS NULL THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Faltan cuentas en cont_configuracion_mdk (CTA_CXC / CTA_VENTAS).';
  END IF;

  -- Datos de la salida
  SELECT DATE(fecha), cliente, nro_documento, moneda,
         IFNULL(NULLIF(tasa_dia,0),1),
         IFNULL(total,0),
         IF(moneda='USD', IFNULL(NULLIF(monto_usd,0), ROUND(IFNULL(total,0)/IFNULL(NULLIF(tasa_dia,0),1),2)), IFNULL(total,0)),
         username
    INTO v_fecha, v_cliente, v_nro, v_moneda, v_tasa, v_total_bs, v_total_trx, v_username
  FROM salidas
  WHERE id = p_salida_id
  LIMIT 1;

  IF v_fecha IS NULL THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Salida no encontrada.';
  END IF;

  -- Evitar duplicar: si ya existe asiento para esa salida y módulo VENTA, no crear otro
  IF EXISTS (
    SELECT 1 FROM cont_asiento_mdk
    WHERE modulo_origen='VENTA' AND origen_tabla='salidas' AND origen_id=p_salida_id AND anulado=0
    LIMIT 1
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Ya existe asiento VENTA para esta salida.';
  END IF;

  -- Crear cabecera de asiento
  INSERT INTO cont_asiento_mdk
  (fecha, descripcion, referencia, modulo_origen, origen_tabla, origen_id, tercero_id,
   moneda_trx, tasa_trx, total_moneda_trx, total_bs, centro_costo_id, username)
  VALUES
  (v_fecha,
   CONCAT('VENTA SALIDA #', p_salida_id),
   v_nro,
   'VENTA',
   'salidas',
   p_salida_id,
   NULL, -- tercero_id lo enlazamos luego cuando tengas cont_tercero_mdk poblado
   v_moneda,
   v_tasa,
   v_total_trx,
   v_total_bs,
   p_centro_costo_id,
   v_username);

  SET v_asiento_id = LAST_INSERT_ID();

  -- Detalle: Debe CxC / Haber Ventas
  INSERT INTO cont_asiento_detalle_mdk
  (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
  VALUES
  (v_asiento_id, v_cta_cxc, p_centro_costo_id, 'CxC por venta', v_moneda, v_tasa, v_total_trx, v_total_bs, 0.00),
  (v_asiento_id, v_cta_ventas, p_centro_costo_id, 'Ingresos por venta', v_moneda, v_tasa, v_total_trx, 0.00, v_total_bs);

  -- Validación Debe/Haber
  IF (
    (SELECT ROUND(SUM(debe_bs - haber_bs),2)
     FROM cont_asiento_detalle_mdk
     WHERE asiento_id = v_asiento_id) <> 0
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Asiento VENTA descuadrado (Debe != Haber).';
  END IF;

END$$

CREATE PROCEDURE sp_mdk_asiento_cobro_cliente(IN p_cobro_id INT UNSIGNED, IN p_centro_costo_id BIGINT)
BEGIN
  DECLARE v_fecha DATE;
  DECLARE v_cliente INT;
  DECLARE v_id_documento INT;
  DECLARE v_username VARCHAR(30);
  DECLARE v_referencia VARCHAR(60);

  DECLARE v_total_bs DECIMAL(18,2);
  DECLARE v_moneda VARCHAR(6);
  DECLARE v_tasa DECIMAL(20,6);

  DECLARE v_es_banco INT DEFAULT 0;
  DECLARE v_asiento_id BIGINT;

  DECLARE v_cta_caja BIGINT;
  DECLARE v_cta_bancos BIGINT;
  DECLARE v_cta_cxc BIGINT;
  DECLARE v_cta_ant_cli BIGINT;

  SET v_cta_caja = fn_cta_id_mdk('CTA_CAJA');
  SET v_cta_bancos = fn_cta_id_mdk('CTA_BANCOS');
  SET v_cta_cxc = fn_cta_id_mdk('CTA_CXC');
  SET v_cta_ant_cli = fn_cta_id_mdk('CTA_ANTICIPO_CLIENTE');

  IF v_cta_caja IS NULL OR v_cta_bancos IS NULL OR v_cta_cxc IS NULL OR v_cta_ant_cli IS NULL THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Faltan cuentas en cont_configuracion_mdk (CTA_CAJA/CTA_BANCOS/CTA_CXC/CTA_ANTICIPO_CLIENTE).';
  END IF;

  SELECT fecha, cliente, id_documento, username, IFNULL(referencia,'')
    INTO v_fecha, v_cliente, v_id_documento, v_username, v_referencia
  FROM cobros_cliente
  WHERE id = p_cobro_id
  LIMIT 1;

  IF v_fecha IS NULL THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro no encontrado.';
  END IF;

  -- Evitar duplicar
  IF EXISTS (
    SELECT 1 FROM cont_asiento_mdk
    WHERE origen_tabla='cobros_cliente' AND origen_id=p_cobro_id AND anulado=0
      AND modulo_origen IN ('COBRO','ANTICIPO_CLI')
    LIMIT 1
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Ya existe asiento para este cobro/anticipo.';
  END IF;

  -- Total en Bs desde detalle
  SELECT IFNULL(ROUND(SUM(IFNULL(monto_bs,0)),2),0),
         IFNULL(MAX(moneda),'Bs'),
         IFNULL(MAX(IFNULL(tasa_moneda,1)),1)
    INTO v_total_bs, v_moneda, v_tasa
  FROM cobros_cliente_detalle
  WHERE cobros_cliente = p_cobro_id;

  IF v_total_bs <= 0 THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro sin detalle o monto_bs=0.';
  END IF;

  -- Caja vs banco: si existe algún banco > 0 => banco
  SELECT COUNT(*) INTO v_es_banco
  FROM cobros_cliente_detalle
  WHERE cobros_cliente = p_cobro_id AND IFNULL(banco,0) > 0;

  -- Crear cabecera
  INSERT INTO cont_asiento_mdk
  (fecha, descripcion, referencia, modulo_origen, origen_tabla, origen_id, tercero_id,
   moneda_trx, tasa_trx, total_moneda_trx, total_bs, centro_costo_id, username)
  VALUES
  (v_fecha,
   IF(v_id_documento=0, CONCAT('ANTICIPO CLIENTE #', p_cobro_id), CONCAT('COBRO CLIENTE #', p_cobro_id)),
   v_referencia,
   IF(v_id_documento=0, 'ANTICIPO_CLI', 'COBRO'),
   'cobros_cliente',
   p_cobro_id,
   NULL,
   v_moneda,
   v_tasa,
   v_total_bs,  -- simplificado: usamos Bs como trx si tu detalle mezcla monedas; lo refinamos luego
   v_total_bs,
   p_centro_costo_id,
   v_username);

  SET v_asiento_id = LAST_INSERT_ID();

  -- Debe: Caja o Bancos
  INSERT INTO cont_asiento_detalle_mdk
  (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
  VALUES
  (v_asiento_id,
   IF(v_es_banco>0, v_cta_bancos, v_cta_caja),
   p_centro_costo_id,
   IF(v_es_banco>0,'Ingreso a bancos','Ingreso a caja'),
   'Bs', 1.000000, v_total_bs,
   v_total_bs, 0.00);

  -- Haber: CxC o Anticipo cliente
  INSERT INTO cont_asiento_detalle_mdk
  (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
  VALUES
  (v_asiento_id,
   IF(v_id_documento=0, v_cta_ant_cli, v_cta_cxc),
   p_centro_costo_id,
   IF(v_id_documento=0,'Anticipo de cliente','Aplicación a CxC'),
   'Bs', 1.000000, v_total_bs,
   0.00, v_total_bs);

  -- Validación
  IF (
    (SELECT ROUND(SUM(debe_bs - haber_bs),2)
     FROM cont_asiento_detalle_mdk
     WHERE asiento_id = v_asiento_id) <> 0
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Asiento COBRO/ANTICIPO descuadrado.';
  END IF;

END$$

CREATE PROCEDURE sp_mdk_asiento_cobro_cliente_detallado(
  IN p_cobro_id INT UNSIGNED,
  IN p_centro_costo_id BIGINT
)
BEGIN
  DECLARE v_fecha DATE;
  DECLARE v_id_documento INT;
  DECLARE v_username VARCHAR(30);
  DECLARE v_ref VARCHAR(60);

  DECLARE v_asiento_id BIGINT;

  DECLARE v_total_bs DECIMAL(18,2);

  -- Cuentas
  DECLARE v_cta_caja BIGINT;
  DECLARE v_cta_bancos BIGINT;
  DECLARE v_cta_cxc BIGINT;
  DECLARE v_cta_ant_cli BIGINT;

  -- Cursor (grupos)
  DECLARE done INT DEFAULT 0;
  DECLARE g_banco INT;
  DECLARE g_metodo VARCHAR(10);
  DECLARE g_moneda VARCHAR(6);
  DECLARE g_tasa DECIMAL(20,6);
  DECLARE g_monto_moneda DECIMAL(18,2);
  DECLARE g_monto_bs DECIMAL(18,2);

  DECLARE cur CURSOR FOR
    SELECT
      IFNULL(banco,0) AS banco,
      IFNULL(metodo_pago,'') AS metodo_pago,
      IFNULL(moneda,'Bs') AS moneda,
      -- tasa: si Bs => 1, si no => tasa_moneda (si viene en NULL, tratamos de derivar de monto_bs / monto_moneda)
      CASE
        WHEN IFNULL(moneda,'Bs') IN ('Bs','VES') THEN 1.000000
        ELSE
          IFNULL(NULLIF(tasa_moneda,0),
            CASE
              WHEN IFNULL(monto_moneda,0) > 0 THEN ROUND(IFNULL(monto_bs,0) / monto_moneda, 6)
              ELSE 1.000000
            END
          )
      END AS tasa_calc,
      -- monto en moneda original (si es Bs y monto_moneda viene NULL, usamos monto_bs como “monto moneda”)
      ROUND(SUM(COALESCE(monto_moneda, monto_bs, 0)), 2) AS monto_moneda_sum,
      ROUND(SUM(IFNULL(monto_bs,0)), 2) AS monto_bs_sum
    FROM cobros_cliente_detalle
    WHERE cobros_cliente = p_cobro_id
    GROUP BY IFNULL(banco,0), IFNULL(metodo_pago,''), IFNULL(moneda,'Bs'),
             CASE
               WHEN IFNULL(moneda,'Bs') IN ('Bs','VES') THEN 1.000000
               ELSE
                 IFNULL(NULLIF(tasa_moneda,0),
                   CASE
                     WHEN IFNULL(monto_moneda,0) > 0 THEN ROUND(IFNULL(monto_bs,0) / monto_moneda, 6)
                     ELSE 1.000000
                   END
                 )
             END;

  DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

  -- Traer cuentas desde config
  SET v_cta_caja    = fn_cta_id_mdk('CTA_CAJA');
  SET v_cta_bancos  = fn_cta_id_mdk('CTA_BANCOS');
  SET v_cta_cxc     = fn_cta_id_mdk('CTA_CXC');
  SET v_cta_ant_cli = fn_cta_id_mdk('CTA_ANTICIPO_CLIENTE');

  IF v_cta_caja IS NULL OR v_cta_bancos IS NULL OR v_cta_cxc IS NULL OR v_cta_ant_cli IS NULL THEN
    SIGNAL SQLSTATE '45000'
      SET MESSAGE_TEXT = 'Faltan cuentas en cont_configuracion_mdk (CTA_CAJA/CTA_BANCOS/CTA_CXC/CTA_ANTICIPO_CLIENTE).';
  END IF;

  -- Cabecera de cobro
  SELECT fecha, id_documento, username, IFNULL(referencia,'')
    INTO v_fecha, v_id_documento, v_username, v_ref
  FROM cobros_cliente
  WHERE id = p_cobro_id
  LIMIT 1;

  IF v_fecha IS NULL THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro no encontrado.';
  END IF;

  -- Evitar duplicados
  IF EXISTS (
    SELECT 1 FROM cont_asiento_mdk
    WHERE origen_tabla='cobros_cliente' AND origen_id=p_cobro_id AND anulado=0
      AND modulo_origen IN ('COBRO','ANTICIPO_CLI')
    LIMIT 1
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Ya existe asiento para este cobro/anticipo.';
  END IF;

  -- Total Bs del cobro (base)
  SELECT ROUND(SUM(IFNULL(monto_bs,0)),2)
    INTO v_total_bs
  FROM cobros_cliente_detalle
  WHERE cobros_cliente = p_cobro_id;

  IF IFNULL(v_total_bs,0) <= 0 THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro sin detalle o monto_bs=0.';
  END IF;

  -- Crear cabecera del asiento (guardamos trx como Bs para cabecera; el detalle tendrá moneda real)
  INSERT INTO cont_asiento_mdk
  (fecha, descripcion, referencia, modulo_origen, origen_tabla, origen_id, tercero_id,
   moneda_trx, tasa_trx, total_moneda_trx, total_bs, centro_costo_id, username)
  VALUES
  (v_fecha,
   IF(v_id_documento=0, CONCAT('ANTICIPO CLIENTE #', p_cobro_id), CONCAT('COBRO CLIENTE #', p_cobro_id)),
   v_ref,
   IF(v_id_documento=0, 'ANTICIPO_CLI', 'COBRO'),
   'cobros_cliente',
   p_cobro_id,
   NULL,
   'Bs', 1.000000,
   v_total_bs,
   v_total_bs,
   p_centro_costo_id,
   v_username);

  SET v_asiento_id = LAST_INSERT_ID();

  -- Insertar detalle por grupos: DEBE (caja/banco) + HABER (CxC o Anticipo cliente) por cada grupo
  OPEN cur;

  read_loop: LOOP
    FETCH cur INTO g_banco, g_metodo, g_moneda, g_tasa, g_monto_moneda, g_monto_bs;
    IF done = 1 THEN
      LEAVE read_loop;
    END IF;

    -- Saltar grupos sin monto
    IF IFNULL(g_monto_bs,0) <= 0 THEN
      ITERATE read_loop;
    END IF;

    -- DEBE: Caja o Bancos (según banco)
    INSERT INTO cont_asiento_detalle_mdk
    (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
    VALUES
    (
      v_asiento_id,
      IF(IFNULL(g_banco,0) > 0, v_cta_bancos, v_cta_caja),
      p_centro_costo_id,
      CONCAT(IF(IFNULL(g_banco,0) > 0,'Ingreso a bancos','Ingreso a caja'),
             ' | metodo:', g_metodo,
             ' | banco:', IFNULL(g_banco,0)),
      g_moneda,
      g_tasa,
      g_monto_moneda,
      g_monto_bs,
      0.00
    );

    -- HABER: CxC o Anticipo Cliente (por el mismo grupo)
    INSERT INTO cont_asiento_detalle_mdk
    (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
    VALUES
    (
      v_asiento_id,
      IF(v_id_documento=0, v_cta_ant_cli, v_cta_cxc),
      p_centro_costo_id,
      CONCAT(IF(v_id_documento=0,'Anticipo cliente','Aplicación a CxC'),
             ' | metodo:', g_metodo,
             ' | banco:', IFNULL(g_banco,0)),
      g_moneda,
      g_tasa,
      g_monto_moneda,
      0.00,
      g_monto_bs
    );

  END LOOP;

  CLOSE cur;

  -- Validación final: Debe = Haber
  IF (
    (SELECT ROUND(SUM(debe_bs - haber_bs),2)
     FROM cont_asiento_detalle_mdk
     WHERE asiento_id = v_asiento_id) <> 0
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Asiento COBRO/ANTICIPO descuadrado (Debe != Haber).';
  END IF;

END$$

  -- Cuentas base desde config
  DECLARE v_cta_caja BIGINT;
  DECLARE v_cta_bancos BIGINT;
  DECLARE v_cta_cxc BIGINT;
  DECLARE v_cta_ant_cli BIGINT;

  -- Cursor (grupos)
  DECLARE done INT DEFAULT 0;
  DECLARE g_banco INT;
  DECLARE g_metodo VARCHAR(10);
  DECLARE g_moneda VARCHAR(6);
  DECLARE g_tasa DECIMAL(20,6);
  DECLARE g_monto_moneda DECIMAL(18,2);
  DECLARE g_monto_bs DECIMAL(18,2);

  -- Mapeo método
  DECLARE m_tipo_destino VARCHAR(15);
  DECLARE m_cuenta_id BIGINT;
  DECLARE m_forzar_destino VARCHAR(10);

  -- Cuenta final DEBE del grupo
  DECLARE v_cuenta_debe BIGINT;

  DECLARE cur CURSOR FOR
    SELECT
      IFNULL(banco,0) AS banco,
      IFNULL(metodo_pago,'') AS metodo_pago,
      IFNULL(moneda,'Bs') AS moneda,
      CASE
        WHEN IFNULL(moneda,'Bs') IN ('Bs','VES') THEN 1.000000
        ELSE
          IFNULL(NULLIF(tasa_moneda,0),
            CASE
              WHEN IFNULL(monto_moneda,0) > 0 THEN ROUND(IFNULL(monto_bs,0) / monto_moneda, 6)
              ELSE 1.000000
            END
          )
      END AS tasa_calc,
      ROUND(SUM(COALESCE(monto_moneda, monto_bs, 0)), 2) AS monto_moneda_sum,
      ROUND(SUM(IFNULL(monto_bs,0)), 2) AS monto_bs_sum
    FROM cobros_cliente_detalle
    WHERE cobros_cliente = p_cobro_id
    GROUP BY IFNULL(banco,0), IFNULL(metodo_pago,''), IFNULL(moneda,'Bs'),
             CASE
               WHEN IFNULL(moneda,'Bs') IN ('Bs','VES') THEN 1.000000
               ELSE
                 IFNULL(NULLIF(tasa_moneda,0),
                   CASE
                     WHEN IFNULL(monto_moneda,0) > 0 THEN ROUND(IFNULL(monto_bs,0) / monto_moneda, 6)
                     ELSE 1.000000
                   END
                 )
             END;

  DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

  -- Cuentas base
  SET v_cta_caja    = fn_cta_id_mdk('CTA_CAJA');
  SET v_cta_bancos  = fn_cta_id_mdk('CTA_BANCOS');
  SET v_cta_cxc     = fn_cta_id_mdk('CTA_CXC');
  SET v_cta_ant_cli = fn_cta_id_mdk('CTA_ANTICIPO_CLIENTE');

  IF v_cta_caja IS NULL OR v_cta_bancos IS NULL OR v_cta_cxc IS NULL OR v_cta_ant_cli IS NULL THEN
    SIGNAL SQLSTATE '45000'
      SET MESSAGE_TEXT = 'Faltan cuentas en cont_configuracion_mdk (CTA_CAJA/CTA_BANCOS/CTA_CXC/CTA_ANTICIPO_CLIENTE).';
  END IF;

  -- Cabecera de cobro
  SELECT fecha, id_documento, username, IFNULL(referencia,'')
    INTO v_fecha, v_id_documento, v_username, v_ref
  FROM cobros_cliente
  WHERE id = p_cobro_id
  LIMIT 1;

  IF v_fecha IS NULL THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro no encontrado.';
  END IF;

  -- Evitar duplicados
  IF EXISTS (
    SELECT 1 FROM cont_asiento_mdk
    WHERE origen_tabla='cobros_cliente' AND origen_id=p_cobro_id AND anulado=0
      AND modulo_origen IN ('COBRO','ANTICIPO_CLI')
    LIMIT 1
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Ya existe asiento para este cobro/anticipo.';
  END IF;

  -- Total Bs
  SELECT ROUND(SUM(IFNULL(monto_bs,0)),2)
    INTO v_total_bs
  FROM cobros_cliente_detalle
  WHERE cobros_cliente = p_cobro_id;

  IF IFNULL(v_total_bs,0) <= 0 THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro sin detalle o monto_bs=0.';
  END IF;

  -- Crear cabecera del asiento (cabecera en Bs; detalle guarda moneda real)
  INSERT INTO cont_asiento_mdk
  (fecha, descripcion, referencia, modulo_origen, origen_tabla, origen_id, tercero_id,
   moneda_trx, tasa_trx, total_moneda_trx, total_bs, centro_costo_id, username)
  VALUES
  (v_fecha,
   IF(v_id_documento=0, CONCAT('ANTICIPO CLIENTE #', p_cobro_id), CONCAT('COBRO CLIENTE #', p_cobro_id)),
   v_ref,
   IF(v_id_documento=0, 'ANTICIPO_CLI', 'COBRO'),
   'cobros_cliente',
   p_cobro_id,
   NULL,
   'Bs', 1.000000,
   v_total_bs,
   v_total_bs,
   p_centro_costo_id,
   v_username);

  SET v_asiento_id = LAST_INSERT_ID();

  -- Detalle por grupos
  OPEN cur;

  read_loop: LOOP
    FETCH cur INTO g_banco, g_metodo, g_moneda, g_tasa, g_monto_moneda, g_monto_bs;
    IF done = 1 THEN
      LEAVE read_loop;
    END IF;

    IF IFNULL(g_monto_bs,0) <= 0 THEN
      ITERATE read_loop;
    END IF;

    -- Buscar mapeo del método
    SET m_tipo_destino = NULL;
    SET m_cuenta_id = NULL;
    SET m_forzar_destino = 'AUTO';

    SELECT tipo_destino, cuenta_id, forzar_destino
      INTO m_tipo_destino, m_cuenta_id, m_forzar_destino
    FROM cont_metodo_cuenta_mdk
    WHERE metodo_pago = g_metodo AND estado = 1
    LIMIT 1;

    IF m_tipo_destino IS NULL THEN
      SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Método de pago sin mapeo en cont_metodo_cuenta_mdk.';
    END IF;

    -- Resolver cuenta DEBE
    SET v_cuenta_debe = NULL;

    IF m_tipo_destino IN ('PASIVO','OTRO') THEN
      -- método especial: debe ir directo a una cuenta fija
      IF m_cuenta_id IS NULL THEN
        SIGNAL SQLSTATE '45000'
          SET MESSAGE_TEXT = 'Método PASIVO/OTRO sin cuenta_id en cont_metodo_cuenta_mdk.';
      END IF;
      SET v_cuenta_debe = m_cuenta_id;

    ELSE
      -- CAJA_BANCO
      IF m_forzar_destino = 'CAJA' THEN
        SET v_cuenta_debe = v_cta_caja;
      ELSEIF m_forzar_destino = 'BANCO' THEN
        SET v_cuenta_debe = v_cta_bancos;
      ELSE
        -- AUTO: decide por banco>0
        SET v_cuenta_debe = IF(IFNULL(g_banco,0) > 0, v_cta_bancos, v_cta_caja);
      END IF;
    END IF;

    -- DEBE (cuenta según método)
    INSERT INTO cont_asiento_detalle_mdk
    (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
    VALUES
    (
      v_asiento_id,
      v_cuenta_debe,
      p_centro_costo_id,
      CONCAT('Cobro metodo:', g_metodo, ' | banco:', IFNULL(g_banco,0)),
      g_moneda,
      g_tasa,
      g_monto_moneda,
      g_monto_bs,
      0.00
    );

    -- HABER (CxC o Anticipo cliente)
    INSERT INTO cont_asiento_detalle_mdk
    (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
    VALUES
    (
      v_asiento_id,
      IF(v_id_documento=0, v_cta_ant_cli, v_cta_cxc),
      p_centro_costo_id,
      CONCAT(IF(v_id_documento=0,'Anticipo','Cobro'), ' aplicado | metodo:', g_metodo),
      g_moneda,
      g_tasa,
      g_monto_moneda,
      0.00,
      g_monto_bs
    );

  END LOOP;

  CLOSE cur;

  -- Validación Debe/Haber
  IF (
    (SELECT ROUND(SUM(debe_bs - haber_bs),2)
     FROM cont_asiento_detalle_mdk
     WHERE asiento_id = v_asiento_id) <> 0
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Asiento COBRO/ANTICIPO descuadrado (Debe != Haber).';
  END IF;

END$$

CREATE PROCEDURE sp_mdk_asiento_cobro_cliente_detallado_fx(
  IN p_cobro_id INT UNSIGNED,
  IN p_centro_costo_id BIGINT
)
BEGIN
  DECLARE v_fecha DATE;
  DECLARE v_id_documento INT;
  DECLARE v_username VARCHAR(30);
  DECLARE v_ref VARCHAR(60);

  DECLARE v_asiento_id BIGINT;
  DECLARE v_total_bs DECIMAL(18,2);

  -- Factura asociada (si aplica)
  DECLARE v_fact_moneda VARCHAR(6);
  DECLARE v_fact_tasa DECIMAL(20,6);

  -- Cuentas base desde config
  DECLARE v_cta_cxc BIGINT;
  DECLARE v_cta_ant_cli BIGINT;
  DECLARE v_cta_dif_gan BIGINT;
  DECLARE v_cta_dif_per BIGINT;
  DECLARE v_cta_caja BIGINT;
  DECLARE v_cta_bancos BIGINT;

  -- Cursor (grupos)
  DECLARE done INT DEFAULT 0;
  DECLARE g_banco INT;
  DECLARE g_metodo VARCHAR(10);
  DECLARE g_moneda VARCHAR(6);

  DECLARE m_forzar_destino VARCHAR(10);

  DECLARE v_cuenta_debe BIGINT;

  -- Para diferencia cambiaria por grupo
  DECLARE v_aplica_cxc_bs DECIMAL(18,2);
  DECLARE v_diff_bs DECIMAL(18,2);

  DECLARE cur CURSOR FOR
    SELECT
      IFNULL(banco,0) AS banco,
      IFNULL(metodo_pago,'') AS metodo_pago,
      IFNULL(moneda,'Bs') AS moneda,
      CASE
        WHEN IFNULL(moneda,'Bs') IN ('Bs','VES') THEN 1.000000
        ELSE
          IFNULL(NULLIF(tasa_moneda,0),
            CASE
              WHEN IFNULL(monto_moneda,0) > 0 THEN ROUND(IFNULL(monto_bs,0) / monto_moneda, 6)
              ELSE 1.000000
            END
          )
      END AS tasa_calc,
      ROUND(SUM(COALESCE(monto_moneda, monto_bs, 0)), 2) AS monto_moneda_sum,
      ROUND(SUM(IFNULL(monto_bs,0)), 2) AS monto_bs_sum
    FROM cobros_cliente_detalle
    WHERE cobros_cliente = p_cobro_id
    GROUP BY IFNULL(banco,0), IFNULL(metodo_pago,''), IFNULL(moneda,'Bs'),
             CASE
               WHEN IFNULL(moneda,'Bs') IN ('Bs','VES') THEN 1.000000
               ELSE
                 IFNULL(NULLIF(tasa_moneda,0),
                   CASE
                     WHEN IFNULL(monto_moneda,0) > 0 THEN ROUND(IFNULL(monto_bs,0) / monto_moneda, 6)
                     ELSE 1.000000
                   END
                 )
             END;

  SET v_cta_ant_cli = fn_cta_id_mdk('CTA_ANTICIPO_CLIENTE');

  SET v_cta_dif_per = fn_cta_id_mdk('CTA_DIF_CAMBIARIA_PER');

  WHERE id = p_cobro_id
  LIMIT 1;

    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro no encontrado.';

  -- Evitar duplicados
  IF EXISTS (
    SELECT 1 FROM cont_asiento_mdk
    WHERE origen_tabla='cobros_cliente' AND origen_id=p_cobro_id AND anulado=0
      AND modulo_origen IN ('COBRO','ANTICIPO_CLI')

  WHERE cobros_cliente = p_cobro_id;

  IF IFNULL(v_total_bs,0) <= 0 THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cobro sin detalle o monto_bs=0.';
  END IF;

  -- Datos de la factura (si aplica)
  SET v_fact_moneda = 'Bs';
  SET v_fact_tasa = 1.000000;

  IF v_id_documento <> 0 THEN
    SELECT IFNULL(moneda,'Bs'), IFNULL(NULLIF(tasa_dia,0),1.000000)
      INTO v_fact_moneda, v_fact_tasa
    FROM salidas
    WHERE id = v_id_documento
    LIMIT 1;
  END IF;

  -- Crear cabecera del asiento
  INSERT INTO cont_asiento_mdk
  (fecha, descripcion, referencia, modulo_origen, origen_tabla, origen_id, tercero_id,
   moneda_trx, tasa_trx, total_moneda_trx, total_bs, centro_costo_id, username)
  VALUES
  (v_fecha,
   IF(v_id_documento=0, CONCAT('ANTICIPO CLIENTE #', p_cobro_id), CONCAT('COBRO CLIENTE #', p_cobro_id)),

   p_centro_costo_id,
   v_username);

  SET v_asiento_id = LAST_INSERT_ID();

  -- Detalle por grupos
  OPEN cur;

  read_loop: LOOP
    FETCH cur INTO g_banco, g_metodo, g_moneda, g_tasa, g_monto_moneda, g_monto_bs;
    IF done = 1 THEN

    END IF;

    IF IFNULL(g_monto_bs,0) <= 0 THEN
      ITERATE read_loop;

    SET m_tipo_destino = NULL;
    SET m_cuenta_id = NULL;
    SET m_forzar_destino = 'AUTO';

    SELECT tipo_destino, cuenta_id, forzar_destino
      INTO m_tipo_destino, m_cuenta_id, m_forzar_destino
    FROM cont_metodo_cuenta_mdk
    WHERE metodo_pago = g_metodo AND estado = 1
    LIMIT 1;

    IF m_tipo_destino IS NULL THEN
      SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Método sin mapeo en cont_metodo_cuenta_mdk.';
    END IF;

    -- Resolver cuenta DEBE
    SET v_cuenta_debe = NULL;

    IF m_tipo_destino IN ('PASIVO','OTRO') THEN
      IF m_cuenta_id IS NULL THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Método PASIVO/OTRO sin cuenta_id en cont_metodo_cuenta_mdk.';
      END IF;
      SET v_cuenta_debe = m_cuenta_id;
    ELSE
      IF m_forzar_destino = 'CAJA' THEN
        SET v_cuenta_debe = v_cta_caja;
      ELSEIF m_forzar_destino = 'BANCO' THEN
        SET v_cuenta_debe = v_cta_bancos;
      ELSE
        SET v_cuenta_debe = IF(IFNULL(g_banco,0) > 0, v_cta_bancos, v_cta_caja);
      END IF;
    END IF;

    -- DEBE: entrada según método
    INSERT INTO cont_asiento_detalle_mdk
    (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
    VALUES
    (v_asiento_id, v_cuenta_debe, p_centro_costo_id,
     CONCAT('Cobro metodo:', g_metodo, ' | banco:', IFNULL(g_banco,0)),
     g_moneda, g_tasa, g_monto_moneda,
     g_monto_bs, 0.00);

    -- Calcular cuánto acreditar a CxC (histórico) cuando factura USD y grupo USD
    SET v_aplica_cxc_bs = g_monto_bs;

    IF v_id_documento <> 0 AND v_fact_moneda = 'USD' AND g_moneda = 'USD' THEN
      SET v_aplica_cxc_bs = ROUND(IFNULL(g_monto_moneda,0) * v_fact_tasa, 2);
    END IF;

    -- HABER: CxC o Anticipo (por el monto aplicado)
    INSERT INTO cont_asiento_detalle_mdk
    (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
    VALUES
    (v_asiento_id,
     IF(v_id_documento=0, v_cta_ant_cli, v_cta_cxc),
     p_centro_costo_id,
     CONCAT(IF(v_id_documento=0,'Anticipo','Cobro'), ' aplicado | metodo:', g_metodo),
     g_moneda, g_tasa, g_monto_moneda,
     0.00, v_aplica_cxc_bs);

    -- Diferencia cambiaria (solo si cobro a factura y factura USD y grupo USD)
    IF v_id_documento <> 0 AND v_fact_moneda = 'USD' AND g_moneda = 'USD' THEN
      SET v_diff_bs = ROUND(g_monto_bs - v_aplica_cxc_bs, 2);

      IF v_diff_bs > 0 THEN
        -- Ganancia: crédito
        INSERT INTO cont_asiento_detalle_mdk
        (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
        VALUES
        (v_asiento_id, v_cta_dif_gan, p_centro_costo_id,
         'Diferencia cambiaria (ganancia)', 'Bs', 1.000000, v_diff_bs, 0.00, v_diff_bs);

      ELSEIF v_diff_bs < 0 THEN
        -- Pérdida: débito
        INSERT INTO cont_asiento_detalle_mdk
        (asiento_id, cuenta_id, centro_costo_id, concepto, moneda_trx, tasa_trx, monto_moneda_trx, debe_bs, haber_bs)
        VALUES
        (v_asiento_id, v_cta_dif_per, p_centro_costo_id,
         'Diferencia cambiaria (pérdida)', 'Bs', 1.000000, ABS(v_diff_bs), ABS(v_diff_bs), 0.00);
      END IF;
    END IF;

  END LOOP;

  CLOSE cur;

  -- Validación final: Debe = Haber
  IF (
    (SELECT ROUND(SUM(debe_bs - haber_bs),2)
     FROM cont_asiento_detalle_mdk
     WHERE asiento_id = v_asiento_id) <> 0
  ) THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Asiento descuadrado (Debe != Haber).';
  END IF;

END$$

CREATE OR REPLACE VIEW view_cont_asiento_resumen_mdk AS
SELECT
    a.id,
    a.fecha,
    a.referencia,
    a.modulo_origen,
    a.origen_id,
    a.descripcion,
    a.anulado,
    a.created_at,

    COALESCE(SUM(d.debe_bs), 0) AS total_debe_bs,
    COALESCE(SUM(d.haber_bs), 0) AS total_haber_bs,
    COALESCE(SUM(d.debe_bs), 0) - COALESCE(SUM(d.haber_bs), 0) AS diferencia_bs,

    CASE
        WHEN a.anulado = 1 THEN 'ANULADO'
        WHEN COALESCE(SUM(d.debe_bs), 0) = COALESCE(SUM(d.haber_bs), 0) THEN 'BALANCEADO'
        ELSE 'DESCUADRADO'
    END AS estado_cuadre

FROM cont_asiento_mdk a
LEFT JOIN cont_asiento_detalle_mdk d
    ON d.asiento_id = a.id
GROUP BY
    a.id,
    a.fecha,
    a.referencia,
    a.modulo_origen,
    a.origen_id,
    a.descripcion,
    a.anulado,
    a.created_at;
